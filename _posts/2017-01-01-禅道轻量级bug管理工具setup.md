# 安装apache,php #

```
yum install -y httpd php php-mysql mysql
wget http://dl.cnezsoft.com/zentao/8.2.5/ZenTaoPMS.8.2.5.zip
unzip ZenTaoPMS.8.2.5.zip 
cd zentaopms/
mv * /var/www/html/
```
